CREATE PROCEDURE [dbo].[pr_Check_Existing_IndivCode]

@code varchar(250), 
@property varchar(250),
@id_indiv int,
@result int OUTPUT

AS 
SET NOCOUNT ON
	

SELECT @result = (
	SELECT 1 WHERE EXISTS (
	SELECT *
  FROM [EcoReleve_ECWP].[dbo].[IndividualDynPropValuesNow] v
  where name = @property
  AND ValueString = @code
  and FK_Individual != @id_indiv
  )
  )

 select @result
 GO






INSERT INTO [dbo].[TVersion] (TVer_FileName,TVer_Date,TVer_DbName) VALUES ('112_pr_Check_Existing_IndivCode',GETDATE(),(SELECT db_name()))


GO
